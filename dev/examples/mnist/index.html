mnist_x = reshape(MNIST.traintensor(Float64), :, 60000);

result = UMAP.fit(mnist_x);
begin
    f = Figure()
    axis = f[1, 1] = Axis(f)
    scatter!(axis, getindex.(result.embedding[1:5000], 1), getindex.(result.embedding[1:5000], 2), color=mnist_y[1:5000], markersize=4)
    f
end
